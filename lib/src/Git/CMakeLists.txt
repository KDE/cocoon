
add_subdirectory( 3rdparty )

)

	3rdparty/dvcsjob.cpp
	3rdparty/gitrunner.cpp
	Blob.cpp
	CloneRepositoryProcess.cpp
	Commit.cpp

	LooseStorage.cpp
